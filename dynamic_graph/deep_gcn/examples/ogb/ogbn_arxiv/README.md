# ogbn-arxiv
## Default 
	--use_gpu False 
	--self_loop False
    --block res+ 	#options: [plain, res, res+]
    --conv gen
    --gcn_aggr max 	#options: [max, mean, add, softmax, softmax_sg, softmax_sum, power, power_sum]
    --num_layers 3

    --norm batch
    --hidden_channels 128
    --epochs 500
    --lr 0.01
